Dateioperationen im Detail

open() + with – Datei lesen

Der with-Block stellt sicher, dass die Datei automatisch geschlossen wird – auch wenn ein Fehler auftritt.

# Datei lesen with open("messung.txt", "r", encoding="utf-8") as datei: inhalt = datei.read() # Alles auf einmal print(inhalt) # Zeilenweise lesen (für große Dateien empfohlen) with open("messung.txt", "r", encoding="utf-8") as datei: for zeile in datei: print(zeile.strip()) # .strip() entfernt \n
open() – Datei schreiben
# Neue Datei schreiben (überschreibt vorhandene!) with open("bericht.txt", "w", encoding="utf-8") as datei: datei.write("Messbericht\n") datei.write(f"Temperatur: {23.5:.2f} °C\n") # An Datei anhängen (existierende Inhalte bleiben erhalten) with open("log.txt", "a", encoding="utf-8") as datei: datei.write("Neue Messung hinzugefügt\n") # Modi: # "r" – lesen (read) # "w" – schreiben (write) – löscht Inhalt! # "a" – anhängen (append)
os.path – Dateipfade
import os # Pfade zusammenbauen (funktioniert auf Windows + Linux) ordner = "daten" dateiname = "sensor_log.csv" pfad = os.path.join(ordner, dateiname) print(pfad) # → daten/sensor_log.csv (Linux) # → daten\sensor_log.csv (Windows) # Prüfen ob Datei existiert if os.path.exists(pfad): print("Datei gefunden!") else: print("Datei nicht gefunden.") # Ordner anlegen (wenn nicht vorhanden) os.makedirs("daten", exist_ok=True)

Spielstand speichern & laden

spielstand.csv lesen und schreiben

import os def spielstand_speichern(schüsse, treffer): with open("spielstand.csv", "w", encoding="utf-8") as f: f.write(f"schuesse,treffer\n{schüsse},{treffer}\n") def spielstand_laden(): if not os.path.exists("spielstand.csv"): return 0, 0 # Standardwerte with open("spielstand.csv", "r", encoding="utf-8") as f: zeilen = f.readlines() daten = zeilen[1].strip().split(",") return int(daten[0]), int(daten[1])
